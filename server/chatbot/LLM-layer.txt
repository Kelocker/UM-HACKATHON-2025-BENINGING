from dotenv import load_dotenv
import os
import json
from openai import OpenAI
from pathlib import Path
import csv


# Fetch the header
folder_path = Path('../data')
csv_headers = {}


csv_files = list(folder_path.glob('*.csv'))
print("Found CSV files:", [file.name for file in csv_files])

for csv_file in csv_files:
    table_name = csv_file.stem
    with open(csv_file, newline='', encoding='utf-8') as f:
        reader = csv.reader(f)
        header = next(reader, None)
        if header:
            cleaned_header = [col if col.strip() else f"{table_name}_id" for col in header]
            csv_headers[table_name] = cleaned_header
            print(f"Cleaned header for '{table_name}': {cleaned_header}")
        else:
            print(f"No header found in {csv_file.name}")


# data = json.dumps(csv_headers, indent=2)




# Load variables from .env file into the environment
load_dotenv()
